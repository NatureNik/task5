const readline = require('readline'); const rl = readline.createInterface({ input: process.stdin, output: process.stdout }); rl.question('Введите номер операции: 1. Сложение 2. Вычитание 3. Умножение: ', (operation) => { switch (operation) { case '1': console.log('Вы выбрали операцию: Сложение'); break; case '2': console.log('Вы выбрали операцию: Вычитание'); break; case '3': console.log('Вы выбрали операцию: Умножение'); break; default: console.log('Операция неопределена'); } rl.close(); });
